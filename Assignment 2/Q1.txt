//BINARY SEARCH

#include <iostream>
using namespace std;
int main()
{
    int arr[6]={1,2,3,11,12,20};
    int num_to_find=20;
    int st=0;
    int end=5;
    while(st<=end){
        int mid=(st+end)/2;
        if(num_to_find>arr[mid]){
            st=mid+1;
        }
        else if(num_to_find<arr[mid]){
            end=mid-1;
        }
        else{
            cout<<"Found at index:"<<mid;
            return 0;
        }
    }
    cout<<"Not found";
    return 0;
}
