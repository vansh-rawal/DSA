// Interleaved Queue

#include <iostream>
using namespace std;

#define size 100

int main() {
    int q[size], n;
    cout<<"Enter number of elements: ";
    cin>>n;
    cout<<"Enter elements: ";
    for(int i=0;i<n;i++){
        cin>>q[i];
    }

    if(n%2!=0){
        cout<<"Queue must have even number of elements"<<endl;
        return 0;
    }

    int half=n/2;
    int first[50], second[50];
    for(int i=0;i<half;i++){
        first[i]=q[i];
    }
    for(int i=0;i<half;i++){
        second[i]=q[half+i];
    }

    int k=0;
    for(int i=0;i<half;i++){
        q[k]=first[i];
        k++;
        q[k]=second[i];
        k++;
    }

    cout<<"Interleaved queue: ";
    for(int i=0;i<n;i++){
        cout<<q[i]<<" ";
    }
    cout<<endl;

    return 0;
}
