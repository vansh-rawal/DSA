// Menu program of Queue

#include <iostream>
using namespace std;

#define size 10
int q[size];
int f=-1, r=-1;

void enqueue(int val){
    if(r==size-1){
        cout<<"Queue is full"<<endl;
        return;
    }
    r++;
    q[r]=val;
    cout<<"Added element: "<<val<<endl;
    if(f==-1){
        f=0;
    }
}

void dequeue(){
    if(f==-1 || f>r){
        cout<<"Queue is empty"<<endl;
        return;
    }
    cout<<"Deleted element: "<<q[f]<<endl;
    f++;
}

void isEmpty(){
    if(f==-1 || f>r){
        cout<<"Queue is empty"<<endl;
    } else {
        cout<<"Queue is not empty"<<endl;
    }
}

void isFull(){
    if(r==size-1){
        cout<<"Queue is full"<<endl;
    } else {
        cout<<"Queue is not full"<<endl;
    }
}

void display(){
    if(f==-1 || f>r){
        cout<<"Queue is empty"<<endl;
        return;
    }
    cout<<"Queue elements: ";
    for(int i=f; i<=r; i++){
        cout<<q[i]<<" ";
    }
    cout<<endl;
}

void peek(){
    if(f==-1 || f>r){
        cout<<"Queue is empty"<<endl;
        return;
    }
    cout<<"Front element: "<<q[f]<<endl;
}

int main(){
    int x,val;
    do{
        cout<<"---MENU---\n";
        cout<<"1. Enqueue\n2. Dequeue\n3. isEmpty\n4. isFull\n5. Display\n6. Peek\n7. Exit\n";
        cout<<"Enter choice: ";
        cin>>x;
        switch(x){
            case 1: {
                cout<<"Value to be added to queue: ";
                cin>>val;
                enqueue(val);
                break;
            }
            case 2: {
                dequeue();
                break;
            }
            case 3: {
                isEmpty();
                break;
            }
            case 4: {
                isFull();
                break;
            }
            case 5: {
                display();
                break;
            }
            case 6: {
                peek();
                break;
            }
            case 7: {
                cout<<"Exit Menu"<<endl;
                break;
            }
            default: {
                cout<<"Invalid choice"<<endl;
            }
        }
    }while(x!=7);
    return 0;
}
